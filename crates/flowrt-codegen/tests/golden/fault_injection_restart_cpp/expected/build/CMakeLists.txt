# FlowRT 管理产物。不要手工修改。
cmake_minimum_required(VERSION 3.22)
project(fault_injection_restart_demo_flowrt_app LANGUAGES C CXX)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_library(fault_injection_restart_demo_flowrt_app INTERFACE)
target_compile_features(fault_injection_restart_demo_flowrt_app INTERFACE cxx_std_20)
target_include_directories(fault_injection_restart_demo_flowrt_app INTERFACE ${CMAKE_CURRENT_LIST_DIR}/../cpp/include)

set(FLOWRT_CPP_RUNTIME_DIR "" CACHE PATH "FlowRT C++ runtime root containing include/flowrt/runtime.hpp")
set(FLOWRT_CXX_COMPILE_OPTIONS "" CACHE STRING "Extra C++ compile options from the FlowRT toolchain profile")
set(FLOWRT_EXE_LINK_OPTIONS "" CACHE STRING "Extra executable link options from the FlowRT toolchain profile")
set(FLOWRT_EXE_LINK_LIBRARIES "" CACHE STRING "Extra executable link libraries from the FlowRT toolchain profile")
if(FLOWRT_CPP_RUNTIME_DIR)
    list(PREPEND CMAKE_PREFIX_PATH "${FLOWRT_CPP_RUNTIME_DIR}")
    list(PREPEND CMAKE_BUILD_RPATH "${FLOWRT_CPP_RUNTIME_DIR}/lib")
endif()
if(NOT FLOWRT_CPP_RUNTIME_DIR)
    find_package(flowrt_runtime 0.24 QUIET)
endif()
option(FLOWRT_ALLOW_REPO_RUNTIME_FALLBACK "Allow falling back to FlowRT source tree runtime/cpp (dev mode only)" OFF)
if(NOT TARGET flowrt::runtime AND NOT FLOWRT_CPP_RUNTIME_DIR AND FLOWRT_ALLOW_REPO_RUNTIME_FALLBACK)
    get_filename_component(_flowrt_repo_runtime "${CMAKE_CURRENT_LIST_DIR}/../../../../runtime/cpp" ABSOLUTE)
    if(EXISTS "${_flowrt_repo_runtime}/include/flowrt/runtime.hpp")
        set(FLOWRT_CPP_RUNTIME_DIR "${_flowrt_repo_runtime}")
    endif()
endif()
if(FLOWRT_CPP_RUNTIME_DIR)
    if(NOT EXISTS "${FLOWRT_CPP_RUNTIME_DIR}/include/flowrt/runtime.hpp")
        message(FATAL_ERROR "FLOWRT_CPP_RUNTIME_DIR does not contain include/flowrt/runtime.hpp: ${FLOWRT_CPP_RUNTIME_DIR}")
    endif()
    target_include_directories(fault_injection_restart_demo_flowrt_app INTERFACE ${FLOWRT_CPP_RUNTIME_DIR}/include)
elseif(TARGET flowrt::runtime)
    target_link_libraries(fault_injection_restart_demo_flowrt_app INTERFACE flowrt::runtime)
else()
    message(FATAL_ERROR "FlowRT C++ runtime was not found. Install the FlowRT package, set CMAKE_PREFIX_PATH, or set FLOWRT_CPP_RUNTIME_DIR to a FlowRT runtime/cpp tree. If developing inside the FlowRT source tree, set -DFLOWRT_ALLOW_REPO_RUNTIME_FALLBACK=ON.")
endif()

add_library(fault_injection_restart_demo_cpp_shell STATIC ../cpp/src/runtime_shell.cpp ../cpp/src/selfdesc.cpp)
target_link_libraries(fault_injection_restart_demo_cpp_shell PUBLIC fault_injection_restart_demo_flowrt_app)

set(FLOWRT_USER_CPP_ROOT "${CMAKE_CURRENT_LIST_DIR}/../../app/cpp")
set(FLOWRT_USER_C_ROOT "${CMAKE_CURRENT_LIST_DIR}/../../app/c")
file(GLOB_RECURSE FLOWRT_DEFAULT_USER_CPP_SOURCES CONFIGURE_DEPENDS
    "${FLOWRT_USER_CPP_ROOT}/*.cpp"
    "${FLOWRT_USER_CPP_ROOT}/*.cc"
    "${FLOWRT_USER_CPP_ROOT}/*.cxx"
    "${FLOWRT_USER_CPP_ROOT}/*.c"
    "${FLOWRT_USER_C_ROOT}/*.c"
)
set(FLOWRT_USER_CPP_SOURCES ${FLOWRT_DEFAULT_USER_CPP_SOURCES} CACHE STRING "User C/C++ sources from app/cpp and app/c that implement flowrt_user::build_app and C callback factories")
if(FLOWRT_USER_CPP_SOURCES)
    add_library(fault_injection_restart_demo_cpp_user STATIC ${FLOWRT_USER_CPP_SOURCES})
    target_include_directories(fault_injection_restart_demo_cpp_user PUBLIC ${FLOWRT_USER_CPP_ROOT} ${FLOWRT_USER_C_ROOT})
    if(FLOWRT_CXX_COMPILE_OPTIONS)
        target_compile_options(fault_injection_restart_demo_cpp_user PRIVATE ${FLOWRT_CXX_COMPILE_OPTIONS})
    endif()
    target_link_libraries(fault_injection_restart_demo_cpp_user PUBLIC fault_injection_restart_demo_flowrt_app)
    add_executable(fault_injection_restart_demo_cpp_app ../cpp/src/main.cpp)
    if(FLOWRT_EXE_LINK_OPTIONS)
        target_link_options(fault_injection_restart_demo_cpp_app PRIVATE ${FLOWRT_EXE_LINK_OPTIONS})
    endif()
    target_link_libraries(fault_injection_restart_demo_cpp_app PRIVATE fault_injection_restart_demo_cpp_shell fault_injection_restart_demo_cpp_user)
    if(FLOWRT_EXE_LINK_LIBRARIES)
        target_link_libraries(fault_injection_restart_demo_cpp_app PRIVATE ${FLOWRT_EXE_LINK_LIBRARIES})
    endif()
endif()

include(CTest)
if(BUILD_TESTING)
    add_executable(fault_injection_restart_demo_message_abi ../cpp/tests/message_abi.cpp)
    target_link_libraries(fault_injection_restart_demo_message_abi PRIVATE fault_injection_restart_demo_flowrt_app)
    set(FLOWRT_ABI_CPP_FIXTURE_DIR "${CMAKE_CURRENT_LIST_DIR}/abi-fixtures/cpp")
    target_compile_definitions(fault_injection_restart_demo_message_abi PRIVATE FLOWRT_ABI_FIXTURE_DIR="${FLOWRT_ABI_CPP_FIXTURE_DIR}")
    if(NOT CMAKE_CROSSCOMPILING)
        add_custom_command(TARGET fault_injection_restart_demo_message_abi POST_BUILD
            COMMAND $<TARGET_FILE:fault_injection_restart_demo_message_abi>
            COMMENT "Generate C++ Message ABI cross-language fixtures")
        add_test(NAME message_abi COMMAND fault_injection_restart_demo_message_abi)
    else()
        message(STATUS "Skipping C++ Message ABI fixture execution while cross compiling")
    endif()
endif()
